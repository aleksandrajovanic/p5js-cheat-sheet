атрибути

background(color);
//постави боју позадине

fill(color);
//постави боју унутрашности објекти

noFill();
//остави необојену унутрашњост

stroke(color);
//постави боју ивице објекта

strokeWeight(weight);
//постави дебљину ивице у пикселима

noStroke();
//не приказуј ивице
